#!/usr/local/bin/perl

# This script finds a string and replaces it with another in all files in a direc
tory.

my $findstring = 'https-princess';
my $replacestring = 'https-qa-stores2-secure';

my $dirname = '.';

opendir (DIR, $dirname) or die "can't opendir $dirname: $!";
while (defined($file = readdir (DIR))) {

    next if $file =~ /^\.\.?$/;   # skip ./ and ../
    my $filename = "$dirname/$file";

    open (FILE, "<$filename") || die ("Can't open file $filename");

    @text = <FILE>;
    close FILE;



    open (FILE, ">$filename") || die ("Can't write file $filename");
    foreach $text ( @text) {
        $text =~ s/$findstring/$replacestring/gs;
        print FILE $text;
    }
    close FILE;

}
closedir(DIR);

exit;
